\def\Figure#1#2#3#4#5{
\begin{figure*}[htp]
\includegraphics[scale=#4,angle=#5]{#1}
\caption{#2}
\label{#3}
\end{figure*}
}

\def\SubFigure#1#2#3#4#5{
\begin{figure*}[htp]
\addtocounter{figure}{-1}
\epsscale{#4}
\includegraphics[angle=#5]{#1}
\caption{#2}
\label{#3}
\end{figure*}
}

\tabletypesize{\footnotesize}

\enddata
\bigskip
#6
\end{deluxetable}
}
